\begin{frame}
\frametitle{Summary}
	\begin{itemize}[<+->]% an approach to creating feedback by embedding actuators, 
	\item Vibrotactile feedback is preferred compared to no feedback.
	\item Vibrotactile feedback can covey information on the way the musical parameters are manipulated. % mention which parameters. not surprisingly the most efficient signals are the once that share similarities with the audio signal
	\end{itemize}
\end{frame}
